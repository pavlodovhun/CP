#Program <_a01>

DATA 
    INT32_t AAAAAAA, BBBBBBB, CYCLEAA, CYCLEBB, XVALUEE;

START
    READ AAAAAAA
    READ BBBBBBB

    FOR CYCLEAA := AAAAAAA TO BBBBBBB DO
        WRITE (CYCLEAA MUL CYCLEAA)

    XVALUEE := 0
    FOR CYCLEAA := 1 TO AAAAAAA DO
        FOR CYCLEBB := 1 TO BBBBBBB DO
            XVALUEE := XVALUEE ADD 1

    WRITE XVALUEE
FINISH
