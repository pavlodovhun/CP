#Program <_a01>

DATA 
    INT32_t AAAAAAA, BBBBBBB, CCCCCCC, MAXMAXM;

START
    READ AAAAAAA
    READ BBBBBBB
    READ CCCCCCC

    IF (AAAAAAA GE BBBBBBB) AND (AAAAAAA GE CCCCCCC) {
        MAXMAXM := AAAAAAA
    } ELSE IF (BBBBBBB GE AAAAAAA) AND (BBBBBBB GE CCCCCCC) {
        MAXMAXM := BBBBBBB
    } ELSE {
        MAXMAXM := CCCCCCC
    }
    WRITE MAXMAXM

    IF (AAAAAAA = BBBBBBB) AND (AAAAAAA = CCCCCCC) AND (BBBBBBB = CCCCCCC) {
        WRITE 1
    } ELSE {
        WRITE 0
    }

    IF !(AAAAAAA GE 0) OR !(BBBBBBB GE 0) OR !(CCCCCCC GE 0) {
        WRITE -1
    } ELSE {
        WRITE 0
    }

    IF (AAAAAAA GE (BBBBBBB ADD CCCCCCC)) {
        WRITE 10
    } ELSE {
        WRITE 0
    }

FINISH
